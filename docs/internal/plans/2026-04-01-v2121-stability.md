def query_kline_batch(self, symbols: list[str], market: str, period: str,
                      start_date: date, end_date: date) -> dict[str, list[Bar]]:
    """Batch query: single SQL with WHERE symbol IN (...)."""
    if not symbols:
        return {}
    table = _safe_table(period)
    placeholders = ",".join(["?"] * len(symbols))
    rows = self._conn.execute(
        f"SELECT * FROM {table} WHERE symbol IN ({placeholders}) AND market=? "
        f"AND time>=? AND time<=? ORDER BY symbol, time",
        [*symbols, market,
         datetime.combine(start_date, datetime.min.time()),
         datetime.combine(end_date, datetime.max.time())],
    ).fetchall()
    result: dict[str, list[Bar]] = {s: [] for s in symbols}
    for r in rows:
        bar = Bar(time=r[0], symbol=r[1], market=r[2], open=r[3], high=r[4],
                  low=r[5], close=r[6], adj_close=r[7], volume=int(r[8]))
        result.setdefault(bar.symbol, []).append(bar)
    return result
```

def get_kline_batch(self, symbols: list[str], market: str, period: str,
                    start_date: date, end_date: date) -> dict[str, list[Bar]]:
    """Batch: single DB query for cached, individual fetch for missing."""
    cached = self._store.query_kline_batch(symbols, market, period, start_date, end_date)
    result = {}
    missing = []
    for sym in symbols:
        bars = cached.get(sym, [])
        if bars:
            cs, ce = bars[0].time.date(), bars[-1].time.date()
            if (cs - start_date).days <= 3 and (end_date - ce).days <= 3:
                result[sym] = bars
                continue
        missing.append(sym)
    # Fetch missing individually (API doesn't support batch)
    for sym in missing:
        bars = self.get_kline(sym, market, period, start_date, end_date)
        if bars:
            result[sym] = bars
    return result
```

"""Gram-Schmidt factor orthogonalization."""
from __future__ import annotations
import numpy as np

def gram_schmidt_orthogonalize(factor_matrix: np.ndarray) -> np.ndarray:
    """Orthogonalize N×K factor matrix via sequential residualization.

    Column 0 unchanged. Column j = residual after regressing on columns 0..j-1.
    NaN rows: excluded from regression, preserved in output.
    """
    T, K = factor_matrix.shape
    if K <= 1:
        return factor_matrix.copy()
    result = factor_matrix.copy()
    for j in range(1, K):
        col = result[:, j].copy()
        valid = ~np.isnan(col)
        for prev_cols in range(j):
            # Only use rows where both current and previous are non-NaN
            prev = result[:, prev_cols]
            both_valid = valid & ~np.isnan(prev)
            if both_valid.sum() < 2:
                continue
            x = prev[both_valid]
            y = col[both_valid]
            # OLS: beta = cov(x,y)/var(x)
            x_mean = x.mean()
            beta = np.dot(x - x_mean, y - y.mean()) / (np.dot(x - x_mean, x - x_mean) + 1e-20)
            col[both_valid] -= beta * x[both_valid]
        result[:, j] = col
    return result
```

"""Index constituent + weight data provider (AKShare + 24h cache)."""
from __future__ import annotations
import logging
import time
from typing import Any

logger = logging.getLogger(__name__)

class IndexDataProvider:
    SUPPORTED_INDICES = {"000300": "沪深300", "000905": "中证500", "000852": "中证1000"}
    _cache: dict[str, tuple[float, Any]] = {}
    _CACHE_TTL = 86400

    def get_constituents(self, index_code: str) -> list[str]:
        cached = self._get_cached(f"cons_{index_code}")
        if cached is not None:
            return cached
        syms = self._fetch_constituents(index_code)
        self._set_cached(f"cons_{index_code}", syms)
        return syms

    def get_weights(self, index_code: str) -> dict[str, float]:
        cached = self._get_cached(f"weights_{index_code}")
        if cached is not None:
            return cached
        constituents = self.get_constituents(index_code)
        if not constituents:
            return {}
        # Try to get actual weights; fallback to equal weight
        weights = self._fetch_weights(index_code, constituents)
        self._set_cached(f"weights_{index_code}", weights)
        return weights

    def _fetch_constituents(self, index_code: str) -> list[str]:
        """Try multiple AKShare APIs with fallback."""
        try:
            import akshare as ak
        except ImportError:
            logger.warning("akshare not installed, cannot fetch index constituents")
            return []
        # Attempt 1
        for func_name, kwargs in [
            ("index_stock_cons_csindex_p", {"symbol": index_code}),
            ("index_stock_cons", {"symbol": index_code}),
        ]:
            try:
                func = getattr(ak, func_name, None)
                if func is None:
                    continue
                df = func(**kwargs)
                if df is not None and not df.empty:
                    # Try common column names for stock codes
                    for col in ["成分券代码", "品种代码", "stock_code", "证券代码", df.columns[0]]:
                        if col in df.columns:
                            codes = df[col].astype(str).tolist()
                            # Convert to standard format: 600519 → 600519.SH
                            return [self._normalize_code(c) for c in codes if c.strip()]
            except Exception as e:
                logger.debug("AKShare %s failed for %s: %s", func_name, index_code, e)
        logger.warning("All AKShare APIs failed for index %s", index_code)
        return []

    def _fetch_weights(self, index_code: str, constituents: list[str]) -> dict[str, float]:
        # Equal weight fallback (AKShare free tier rarely has weights)
        n = len(constituents)
        return {s: 1.0 / n for s in constituents} if n > 0 else {}

    def _normalize_code(self, code: str) -> str:
        code = code.strip().split(".")[0]  # remove existing suffix
        if code.startswith("6"):
            return f"{code}.SH"
        return f"{code}.SZ"

    def _get_cached(self, key: str):
        entry = self._cache.get(key)
        if entry and time.monotonic() - entry[0] < self._CACHE_TTL:
            return entry[1]
        return None

    def _set_cached(self, key: str, value):
        self._cache[key] = (time.monotonic(), value)
```
