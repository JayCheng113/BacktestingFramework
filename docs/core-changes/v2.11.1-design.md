## 一、compute_raw() 接口扩展（C1 修正，全局基础设施）

### 问题

当前所有 CrossSectionalFactor.compute() 返回百分位排名 [0,1]。中性化和因子合成需要**原始值**（如 EP=0.2、收益率=0.05），不是排名。对排名做中性化会产生错误结果。

### 方案

修改 `ez/portfolio/cross_factor.py`（非 CORE 文件）：

```python
class CrossSectionalFactor(ABC):
    # ... 现有代码不变 ...

    def compute_raw(self, universe_data, date) -> pd.Series:
        """返回原始得分（不排名）。中性化和因子合成使用此方法。

        默认实现等同 compute()（向后兼容已有子类）。
        新因子应覆盖此方法返回原始值，compute() 则调用 compute_raw() 再排名。
        """
        return self.compute(universe_data, date)
```

### 需要重构的因子类

**1. MomentumRank / VolumeRank / ReverseVolatilityRank** (cross_factor.py)：

```python
class MomentumRank(CrossSectionalFactor):
    def compute_raw(self, universe_data, date) -> pd.Series:
        scores = {}
        for sym, df in universe_data.items():
            if len(df) < self._period or "adj_close" not in df.columns:
                continue
            close = df["adj_close"]
            ret = (close.iloc[-1] - close.iloc[-self._period]) / close.iloc[-self._period]
            scores[sym] = ret
        return pd.Series(scores) if scores else pd.Series(dtype=float)

    def compute(self, universe_data, date) -> pd.Series:
        raw = self.compute_raw(universe_data, date)
        return raw.rank(pct=True) if len(raw) > 0 else raw
```

同样重构 VolumeRank、ReverseVolatilityRank（提取原始值到 compute_raw，compute 调 raw 再 rank）。

**2. FundamentalCrossFactor** (ez/factor/builtin/fundamental.py)：

```python
class FundamentalCrossFactor(CrossSectionalFactor):
    def compute_raw(self, universe_data, date) -> pd.Series:
        """原始得分，方向已处理（higher_is_better=False 的已取负）。"""
        if self._store is None:
            return pd.Series(dtype=float)
        symbols = list(universe_data.keys())
        d = date.date() if hasattr(date, 'date') else date
        raw = self._raw_scores(symbols, d)
        if not raw:
            return pd.Series(dtype=float)
        s = pd.Series(raw)
        if not self.higher_is_better:
            s = -s
        return s

    def compute(self, universe_data, date) -> pd.Series:
        raw = self.compute_raw(universe_data, date)
        return raw.rank(pct=True) if len(raw) > 0 else raw
```

**3. AmihudIlliquidity 和 IndustryMomentum**（覆盖了 compute 的特殊因子）：
同样提取 compute_raw()，compute() 调 raw 再 rank。

### 向后兼容

- 用户自定义的 CrossSectionalFactor 子类：只实现了 compute()，没有 compute_raw()
- 默认 compute_raw() 返回 compute() 结果（已排名的值）→ 对中性化效果打折但不报错
- 这是 acceptable degradation，不是 breaking change

## 二、F2 行业中性化

### 新文件: `ez/portfolio/neutralization.py`

```python
def neutralize_by_industry(
    raw_scores: pd.Series,           # compute_raw() 的输出，原始值
    industry_map: dict[str, str],    # symbol → industry
    min_coverage: float = 0.5,       # 行业覆盖率阈值
) -> tuple[pd.Series, list[str]]:
    """行业中性化：因子值减去同行业均值。

    Args:
        raw_scores: 原始因子得分（未排名）
        industry_map: 股票→行业映射
        min_coverage: 有行业标签的股票占比低于此值时跳过中性化

    Returns:
        (中性化后的得分, 警告列表)

    警告情况:
        - 覆盖率 < min_coverage → 返回原值 + 警告
        - 行业内仅 1 只 → 该股票得分设为 NaN（无参照）
    """
    warnings = []

    # 1. 检查覆盖率
    labeled = sum(1 for s in raw_scores.index if s in industry_map and industry_map[s])
    coverage = labeled / len(raw_scores) if len(raw_scores) > 0 else 0
    if coverage < min_coverage:
        warnings.append(f"行业覆盖率仅 {coverage:.0%}，跳过中性化（需 ≥ {min_coverage:.0%}）")
        return raw_scores, warnings

    # 2. 分配行业
    industries = pd.Series({s: industry_map.get(s, '') for s in raw_scores.index})

    # 3. 计算行业均值并中性化
    df = pd.DataFrame({'score': raw_scores, 'industry': industries})
    df.loc[df['industry'] == '', 'industry'] = pd.NA  # 无行业的不参与分组

    # 行业均值（忽略 NaN 和无行业的）
    industry_mean = df.groupby('industry')['score'].transform('mean')
    industry_count = df.groupby('industry')['score'].transform('count')

    neutralized = df['score'] - industry_mean

    # 行业内仅 1 只 → 设 NaN
    neutralized[industry_count <= 1] = pd.NA

    # 无行业的保留原值
    no_industry = df['industry'].isna()
    neutralized[no_industry] = df.loc[no_industry, 'score']

    if no_industry.sum() > 0:
        warnings.append(f"{no_industry.sum()} 只股票无行业标签，保留原始值")

    return neutralized.dropna(), warnings
```

## 三、F3 AlphaCombiner

### 新文件: `ez/portfolio/alpha_combiner.py`

```python
class AlphaCombiner(CrossSectionalFactor):
    """多因子合成。将 N 个子因子的原始得分 z-score 标准化后加权求和。

    不走 __init_subclass__ 自动注册（需要子因子参数，无法无参实例化）。
    在 _create_strategy() 中特殊处理。

    三种合成方法:
      - equal: 等权平均
      - ic: 用各因子的历史 IC 均值作为权重
      - icir: 用各因子的 ICIR 作为权重

    IC/ICIR 权重由 API 层预计算（用回测开始前的数据），作为固定 weights 传入。
    不在 compute() 中滚动计算（P1 修正: 避免性能问题和复杂度）。
    """

    # 阻止自动注册
    __abstractmethods__ = frozenset()  # 非抽象，但不让 __init_subclass__ 注册

    def __init__(self, factors: list, weights: dict[str, float] | None = None):
        """
        Args:
            factors: 子因子列表（CrossSectionalFactor 实例）
            weights: {factor.name: weight}。None → 等权。
                     由 API 层预计算 IC/ICIR 后传入。
        """
        self._factors = factors
        self._weights = weights

    @property
    def name(self) -> str:
        names = '+'.join(f.name for f in self._factors)
        return f"alpha({names})"

    @property
    def description(self) -> str:
        return "多因子合成"

    @property
    def warmup_period(self) -> int:
        return max((f.warmup_period for f in self._factors), default=0)

    def compute_raw(self, universe_data, date) -> pd.Series:
        """合成原始得分（z-score 加权求和，未排名）。

        C2 修正: std < 1e-10 时 z-score 全 0（避免除零）
        C3 修正: 缺失因子按可用因子重新归一化权重（不排除股票）
        E2 修正: 每个子因子只调一次 compute_raw()
        """
        import numpy as np

        # E2: 每个子因子只调一次
        raw_scores = {}
        for f in self._factors:
            scores = f.compute_raw(universe_data, date)
            if len(scores) > 0:
                raw_scores[f.name] = scores

        if not raw_scores:
            return pd.Series(dtype=float)

        # Z-score 标准化
        z_df = pd.DataFrame(raw_scores)
        for col in z_df.columns:
            s = z_df[col]
            std = s.std()
            if std > 1e-10:  # C2: 除零保护
                z_df[col] = (s - s.mean()) / std
            else:
                z_df[col] = 0.0

        # 加权求和
        if self._weights:
            w = pd.Series({col: self._weights.get(col, 0.0) for col in z_df.columns})
        else:
            w = pd.Series(1.0, index=z_df.columns)

        # C3: 缺失值处理 — 按可用因子重新归一化
        # 分子: 有值因子的加权和
        # 分母: 有值因子的权重和
        weighted = z_df.mul(w)
        available_weight = z_df.notna().astype(float).mul(w)

        numerator = weighted.sum(axis=1, skipna=True)
        denominator = available_weight.sum(axis=1)
        denominator = denominator.replace(0, np.nan)

        combined = numerator / denominator
        return combined.dropna()

    def compute(self, universe_data, date) -> pd.Series:
        raw = self.compute_raw(universe_data, date)
        return raw.rank(pct=True) if len(raw) > 0 else raw
```

### 阻止自动注册的方法

问题: `__init_subclass__` 检查 `not getattr(cls, '__abstractmethods__', None)`。AlphaCombiner 不是抽象类，会被注册，但无法无参实例化。

方案: 在 AlphaCombiner 定义后，从 registry 中移除：

```python
# alpha_combiner.py 末尾
CrossSectionalFactor._registry.pop("AlphaCombiner", None)
```

这比 hack `__abstractmethods__` 更清晰。

```python
def _compute_alpha_weights(factors, symbols, market, start, end, method) -> dict[str, float]:
    """用回测开始前的数据预计算 IC/ICIR 权重。

    P1 修正: 一次性计算，不滚动。
    使用 start 之前 1 年的数据作为训练窗口。
    """
    from ez.portfolio.cross_evaluator import evaluate_cross_sectional_factor
    from datetime import timedelta

    train_end = start - timedelta(days=1)
    train_start = start - timedelta(days=365)

    universe_data, calendar = _fetch_data(symbols, market, train_start, train_end, lookback_days=300)

    weights = {}
    for f in factors:
        try:
            result = evaluate_cross_sectional_factor(
                factor=f, universe_data=universe_data, calendar=calendar,
                start=train_start, end=train_end, forward_days=5, eval_freq='weekly',
            )
            if method == 'ic':
                # 保留符号: 负IC=因子方向错→负权重→被抑制 (审计修正: 去掉abs)
                weights[f.name] = result.mean_ic if result.mean_ic else 0.0
            elif method == 'icir':
                weights[f.name] = result.icir if result.icir else 0.0
        except Exception as e:
            logger.warning("AlphaCombiner weight failed for %s: %s", f.name, e)
            weights[f.name] = 0.0

    # 所有权重都为0时回退等权 (审计修正: 用all而非sum防止正负抵消)
    if all(v == 0 for v in weights.values()):
        return None  # fallback to equal weight
    return weights
```

```python
@router.post("/search")
def portfolio_search(req: PortfolioSearchRequest):
    """批量参数搜索。数据只取一次，循环运行回测，按夏普排名。"""
    end = req.end_date or date.today()

    # 3. 基本面数据预加载一次（E1 修正）
    _preload_fundamental_if_needed(combos, req.symbols, start, end)

    # 4. 构建共享 cost_model
    cost_model = CostModel(...)

    # 5. 循环运行
    results = []
    for i, params in enumerate(combos):
        try:
            strategy, _ = _create_strategy(req.strategy_name, params)
            # _create_strategy 内不再重复 preload（已在步骤 3 完成）

            result = run_portfolio_backtest(
                strategy=strategy, universe=Universe(req.symbols),
                universe_data=universe_data, calendar=calendar,
                start=start, end=end, freq=req.freq,
                initial_cash=req.initial_cash, cost_model=cost_model,
                lot_size=req.lot_size, limit_pct=req.limit_pct,
            )

            # P5 修正: 只返回关键指标，不返回 equity_curve
            results.append({
                "rank": 0,  # 后面排序填入
                "params": params,
                "sharpe": result.metrics.get("sharpe_ratio"),
                "total_return": result.metrics.get("total_return"),
                "max_drawdown": result.metrics.get("max_drawdown"),
                "trade_count": result.metrics.get("trade_count", 0),
                "annualized_return": result.metrics.get("annualized_return"),
            })
        except Exception as e:
            logger.warning("Search combo %d failed: %s", i, e)
            continue

    # 6. 排名 (审计修正: 0.0不当缺失, NaN/None排末尾)
    def _sort_key(r):
        v = r.get("sharpe")
        if v is None or (isinstance(v, float) and (np.isnan(v) or np.isinf(v))):
            return -999.0
        return v
    results.sort(key=_sort_key, reverse=True)
    for i, r in enumerate(results):
        r["rank"] = i + 1

    return {
        "results": results,
        "total_combinations": len(combos),
        "completed": len(results),
    }
```

```python
def _generate_combinations(param_grid: dict, max_combos: int) -> list[dict]:
    """网格展开。超过 max_combos 则随机采样。"""
    import itertools
    keys = list(param_grid.keys())
    values = [param_grid[k] for k in keys]
    all_combos = [dict(zip(keys, v)) for v in itertools.product(*values)]
    if len(all_combos) > max_combos:
        import random
        random.shuffle(all_combos)
        all_combos = all_combos[:max_combos]
    return all_combos


def _preload_fundamental_if_needed(combos, symbols, start, end):
    """检查参数组合中是否用了基本面因子，统一 preload 一次。
    审计修正: 覆盖 combo["factor"] + combo["factors"] + combo["alpha_factors"] 三条路径。
    """
    from ez.factor.builtin.fundamental import FundamentalCrossFactor, NEEDS_FINA
    need_fina = False
    needs_preload = False
    factor_map = _get_factor_map()

    def _check(fn):
        nonlocal needs_preload, need_fina
        factory = factor_map.get(fn)
        if factory and isinstance(factory, type) and issubclass(factory, FundamentalCrossFactor):
            needs_preload = True
            if factory.__name__ in NEEDS_FINA:
                need_fina = True

    for combo in combos:
        _check(combo.get("factor", ""))
        if combo.get("factor") == "alpha_combiner":
            for sf in combo.get("alpha_factors", []):
                _check(sf)
        for mfn in combo.get("factors", []):
            _check(mfn)

    if needs_preload:
        _ensure_fundamental_data(symbols, start, end, need_fina)
```
